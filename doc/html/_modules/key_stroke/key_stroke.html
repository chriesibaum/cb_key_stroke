        """
        Checks whether a button from the keys list has been pressed.

        Args:
            keys (list):    List of keys to check

        Return:
            Bool:       True if a key was pressed, False if not
        """
        if self.kbhit():
            c = self.getch()
            if c in keys:         # x1b is ESC
                return True

        return False
